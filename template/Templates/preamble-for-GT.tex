% --- Document Class & Basic Setup ---
\documentclass[12pt]{report}

% --- Geometry & Page Layout ---
\usepackage[a4paper, total={170mm,257mm}, left=20mm, top=20mm, right=20mm, bottom=23mm]{geometry}
\usepackage{pgfpages}
\pgfpagesdeclarelayout{boxed}{
    \edef\pgfpageoptionborder{0pt}
}{
    \pgfpagesphysicalpageoptions{logical pages=1}
    \pgfpageslogicalpageoptions{1}{
        border code=\pgfsetlinewidth{2pt}\pgfstroke,
        border shrink=\pgfpageoptionborder,
        resized width=.95\pgfphysicalwidth,
        resized height=.95\pgfphysicalheight,
        center=\pgfpoint{.5\pgfphysicalwidth}{.5\pgfphysicalheight}
    }
}
\pgfpagesuselayout{boxed}

% --- Graphics & Figures ---
\usepackage{tikz, pgfplots}
\usetikzlibrary{positioning, shapes, arrows}
\pgfplotsset{compat=1.18}
\usepackage{graphicx, float, subfig, wrapfig}

% --- Math & Equations ---
\usepackage{amsmath, amssymb, empheq}
\usepackage{mathtools} % Optional (enhances amsmath)

% --- Sections & Formatting ---
\usepackage[Glenn]{fncychap}
\usepackage{sectsty}

\subsectionfont{\Large\color[HTML]{AE0000}}
\subsubsectionfont{\large\color[HTML]{123175}}

% --- Tables ---
\usepackage{caption, colortbl, booktabs}

% --- Hyperlinks (load near the end) ---
\usepackage{hyperref}
\hypersetup{
    colorlinks=true,
    linkcolor=violet,
    urlcolor=red,
}

% --- Page Style & Footer ---
\usepackage{fancyhdr}

\cfoot{\thepage}
